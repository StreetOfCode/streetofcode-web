import { motion } from "framer-motion";
import { useEffect, useState } from "react";

export const AnimatedCodeEditor = () => {
  const [codeLines, setCodeLines] = useState<string[]>([
    "// vytvor funkciu na sčítanie",
    "",
    "",
    "",
  ]);

  useEffect(() => {
    const lines = [
      "// vytvor funkciu na sčítanie",
      "function sum(a: number, b: number) {",
      "  return a + b;",
      "}",
    ];

    let currentLine = 1;
    let currentChar = 0;

    const typeCode = () => {
      if (currentLine >= lines.length) {
        setTimeout(() => {
          setCodeLines(["// vytvor funkciu na sčítanie", "", "", ""]);
          currentLine = 1;
          currentChar = 0;
          setTimeout(typeCode, 1000);
        }, 3000);
        return;
      }

      const targetLine = lines[currentLine];
      if (currentChar <= targetLine.length) {
        const newLines = [...lines];
        newLines[currentLine] = targetLine.substring(0, currentChar);
        setCodeLines(newLines);
        currentChar++;
        setTimeout(typeCode, 50);
      } else {
        currentLine++;
        currentChar = 0;
        setTimeout(typeCode, 200);
      }
    };

    const timer = setTimeout(typeCode, 1000);
    return () => clearTimeout(timer);
  }, []);

  return (
    <motion.div
      className="relative rounded-xl overflow-hidden shadow-2xl"
      animate={{ y: [0, -10, 0] }}
      transition={{ duration: 6, repeat: Infinity, ease: "easeInOut" }}
    >
      {/* Editor header */}
      <div className="bg-card border border-border px-4 py-3 flex items-center gap-2">
        <div className="flex gap-2">
          <div className="w-3 h-3 rounded-full bg-destructive" />
          <div className="w-3 h-3 rounded-full bg-yellow-500" />
          <div className="w-3 h-3 rounded-full bg-green-500" />
        </div>
        <span className="text-xs text-muted-foreground ml-4">sum.ts</span>
      </div>

      {/* Code content */}
      <div className="bg-card/50 backdrop-blur-sm border-x border-b border-border p-6 font-mono text-sm">
        {codeLines.map((line, i) => (
          <div key={i} className="flex gap-4 min-h-[24px]">
            <span className="text-muted-foreground select-none w-6 text-right">{i + 1}</span>
            <span className="flex-1">
              {i === 0 && (
                <span className="text-muted-foreground italic">{line}</span>
              )}
              {i === 1 && (
                <span>
                  <span className="text-secondary">function</span>
                  {" "}
                  <span className="text-accent">sum</span>
                  <span className="text-muted-foreground">(</span>
                  <span className="text-foreground">a</span>
                  <span className="text-muted-foreground">: </span>
                  <span className="text-secondary">number</span>
                  <span className="text-muted-foreground">, </span>
                  <span className="text-foreground">b</span>
                  <span className="text-muted-foreground">: </span>
                  <span className="text-secondary">number</span>
                  <span className="text-muted-foreground">) {"{"}</span>
                  {line.length < 42 && <span className="animate-pulse">|</span>}
                </span>
              )}
              {i === 2 && (
                <span>
                  {"  "}
                  <span className="text-primary">return</span>
                  {" "}
                  <span className="text-foreground">a + b</span>
                  <span className="text-muted-foreground">;</span>
                  {line.length < 15 && <span className="animate-pulse">|</span>}
                </span>
              )}
              {i === 3 && (
                <span>
                  <span className="text-muted-foreground">{"}"}</span>
                  {line.length < 1 && <span className="animate-pulse">|</span>}
                </span>
              )}
            </span>
          </div>
        ))}
      </div>

      {/* Glow effect */}
      <div className="absolute inset-0 bg-gradient-to-br from-primary/10 via-transparent to-accent/10 pointer-events-none" />
    </motion.div>
  );
};

export const ThemeToggle = () => {
  const [theme, setTheme] = useState<"light" | "dark">("dark");

  useEffect(() => {
    const savedTheme = localStorage.getItem("theme") as "light" | "dark" | null;
    const prefersDark = window.matchMedia("(prefers-color-scheme: dark)").matches;
    const initialTheme = savedTheme || (prefersDark ? "dark" : "light");
    
    setTheme(initialTheme);
    document.documentElement.classList.toggle("dark", initialTheme === "dark");
  }, []);

  const toggleTheme = () => {
    const newTheme = theme === "dark" ? "light" : "dark";
    setTheme(newTheme);
    localStorage.setItem("theme", newTheme);
    document.documentElement.classList.toggle("dark", newTheme === "dark");
  };

  return (
    <Button
      variant="outline"
      size="icon"
      onClick={toggleTheme}
      className="fixed top-4 right-4 z-50 rounded-full"
      aria-label="Toggle theme"
    >
      {theme === "dark" ? (
        <Sun className="h-5 w-5" />
      ) : (
        <Moon className="h-5 w-5" />
      )}
    </Button>
  );
};
